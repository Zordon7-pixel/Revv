From backend directory:
  npm run seed:demo
  npm run seed:demo:force

Via bash:
  bash scripts/seed-demo.sh
  bash scripts/seed-demo.sh --force

Direct Node:
  cd backend && node ../scripts/seed-demo.js
  cd backend && node ../scripts/seed-demo.js --force
